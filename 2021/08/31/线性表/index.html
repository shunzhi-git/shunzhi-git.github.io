线性表

发表于 2021-08-31 更新于 2021-10-30 分类于 数据结构 阅读次数： 阅读次数： Valine：

一、线性表相关概念

1.线性表的定义：

零个或多个元素的有限序列。

2. 线性表的存储结构：

  • 顺序存储结构
  • 链式存储结构：单链表、静态链表、双链表、循环链表。
  • 顺序和链式存储结构不同点主要在地址是否连续，这个特点也决定了其各种操作的算法复杂度。

3. 线性表的操作

  • 整表创建，整表删除，元素获取，元素插入，元素删除

二、线性表的顺序存储结构：

用一段连续的存储单元依次存储线性表的数据元素，数组、字符串都属于顺序存储结构。

/*
*定义函数返回值
*/
#define OK           (0)
#define ERROR        (-1)
#define ELEM_MAX_NUM (10)

/*
*线性表结构体类型
*/
struct list
{
        int length;
        int data[ELEM_MAX_NUM];
};

/*
*获取顺序结构线性表L的第i个元素。
*/
int get_list_elem(list L, int i, int *e)
{
        if (L.length == 0 || i >= L.length )
                return ERROR;

        *e = L.data[i];
        return OK;
}

/*
*插入顺序结构线性表L的第i个元素。
*/
int insert_list_elem(list L, int i, int e)
{
        int k;

        if (L.length == 0 || i >= L.length)
                return ERROR;

        for (int k = L.length; k > i; k--)
        {
                L.data[k] = L.data[k-1]
        }

        L.length++;
        L.data[k] = e;

        return OK;
}

/*
*删除顺序结构线性表L的第i个元素。
*/
int delete_list_elem(list L, int i)
{
        int k;

        if (L.length == 0 || i >= L.length)
                return ERROR;

        for (int k = i; k < L.length; k++)
        {
                L.data[k] = L.data[k+1]
        }

        L.length--;

        return OK;
}

/*
*输入字符串原地逆置。
*/
int sort_string(char *string)
{
        int len,i;
        char *start = string;
        char *end = string - 1;
        char p;

        while (start < end)
        {
                p = *start;
                *start = *end;
                *end = p;
                start++;
                end--;
        }

        return OK;
}

{

        if(*string != '\0')

                sort_string(string++)
                printf("%c\n", *string);
        }
        
        return OK;
}

三、单链表存储结构：

  • 节点（node）：其由数据域和指针域两部分组成。
  • 链表：多个节点链接成一个线性表。
  • 单链表：每一个节点指包含一个后继指针域的链表成为单链表。
  • 头指针：链表中第一个节点存储的位置叫做头指针。
  • 头节点：单链表第一个节点前附设一个节点，成为头节点，主要方便第一个节点的插入和删除，非必需要素。

/*
*定义函数返回值
*/
#define OK           (0)
#define ERROR        (-1)

/*
*线性表结构体类型
*/
struct node
{
        int          data;
        struct node *next;
}node;

typedef node *link_list_node;

/*
*获取单链表L的第n个元素。
*/
int get_singly_link_list_elem(link_list_node L, int n, int *e)
{
        int i = 1
        link_list_node p;

        p = L->next;

        while(p && i < n)
        {
                p = p->next;
                i++;
        }

        if(!p)
        {
                return ERROR;
        }

        *e = p->data;
        
        return OK;
}

/*
*单链表L插入第n个元素。
*/
int insert_singly_link_list_elem(link_list_node L, int n, int e)
{
        int i = 1;
        link_list_node p,s;

        p = L->next;
        while (p && i < n)
        {
                p = p->next;
                i++;
        }

        if(!p)
        {
                return ERROR;
        }

        s = malloc(sizeof(node));
        s->next = p->next;
        s->data = e;
        p->netx = s;

        return OK;
}

/*
*删除单链表L的第n个元素。
*/
int delete_singly_link_list_elem(list L, int n)
{
        int i = 1;
        link_list_node p,d;

        p = L->next;
        while (p && i < n)
        {
            p = p->next;    
        }

        if(!p)
        {
                return ERROR;
        }

        d = p->next;
        p->next = d->next;
        free(d);

        return OK;
}

四、循环链表和双向链表存储结构：

  • 循环链 表较单链表多了一个指向链表首地址的指针，即尾节点指针指向头节点。如此链表的任意节点都可以直接遍历整个链表的全部节点。
  • 双向链表
  • 其指针域有前驱和后继两个指针组成，前驱指向前一个节点，后继指向后一个节点。如此查找上一个节点，算法复杂度就变成了O(1)。
  • 需要特别注意的是在操作节点的时候要同时改变前驱和后继两个指针的变量。
  • 双向链表其本质是通过通过空间换取时间，提高算法性能。
